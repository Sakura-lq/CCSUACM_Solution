#include <bits/stdc++.h>
using namespace std;
#define int long long
#define double long double
#define endl '\n'

typedef struct
{
    int l, r, h;
} Node;

int d[40005], v[40005], b[40005];

void build(int s, int t, int p)
{
    if (s == t)
    {
        d[p] = 0;
        return;
    }
    int m = s + ((t - s) >> 1);
    build(s, m, p << 1);
    build(m + 1, t, (p << 1) + 1);
    d[p] = max(d[p << 1], d[(p << 1) + 1]);
}

void push(int p, int s, int t)
{
    if (v[p])
    {
        int m = s + ((t - s) >> 1);
        d[p << 1] = max(d[p << 1], b[p]);
        d[(p << 1) + 1] = max(d[(p << 1) + 1], b[p]);
        b[p << 1] = max(b[p << 1], b[p]);
        b[(p << 1) + 1] = max(b[(p << 1) + 1], b[p]);
        v[p << 1] = v[(p << 1) + 1] = 1;
        v[p] = 0;
    }
}

void update(int l, int r, int s, int t, int p, int c)
{

    if (l <= s && t <= r)
    {
        d[p] = max(d[p], c);
        b[p] = max(b[p], c);
        v[p] = 1;
        return;
    }
    push(p, s, t);
    int m = s + ((t - s) >> 1);
    if (l <= m)
    {
        update(l, r, s, m, p << 1, c);
    }
    if (r > m)
    {
        update(l, r, m + 1, t, (p << 1) + 1, c);
    }
    d[p] = max(d[p << 1], d[(p << 1) + 1]);
}

int query(int pos, int s, int t, int p)
{
    if (s == t)
    {
        return d[p];
    }
    push(p, s, t);
    int m = s + ((t - s) >> 1);
    if (pos <= m)
    {
        return query(pos, s, m, p << 1);
    }
    else
    {
        return query(pos, m + 1, t, (p << 1) + 1);
    }
}

void solve()
{
    int n;
    cin >> n;
    vector<Node> date(n);
    int mx = 0;

    for (int i = 0; i < n; i++)
    {
        int l, r, h;
        cin >> h >> l >> r;
        date[i] = {l, r, h};
        mx = max(mx, r);
    }

    build(1, mx, 1);

    sort(date.begin(), date.end(), [&](const Node &a, const Node &b){
        if(a.h != b.h) return a.h < b.h;
        return a.l < b.l; 
    });

    int ans = 0;
    for (int i = 0; i < n; i++)
    {
        auto &[l, r, h] = date[i];
        ans += h - query(l, 1, mx, 1);
        ans += h - query(r, 1, mx, 1);
        if (r - l > 1)
        {
            update(l + 1, r - 1, 1, mx, 1, h);
        }
    }

    cout << ans << endl;
}

signed main()
{
    int T = 1;
    // cin >> T;
    while (T--)
    {
        solve();
    }
    return 0;
}
